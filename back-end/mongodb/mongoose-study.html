# mongoose入门

时间：2018-09-13

标签：#mongodb

# 1.mongoose连接mongodb

这里采用express框架进行mongoose的操作，目录结构为：

project
  |--bin
  |--models
     |--mongo.js
     |--user.js
  |--public
     |--images
     |--javascripts
     |--stylesheets
  |--routes
     |--api.js
     |--index.js
  |--views
     |--index.html
     |--error.html
     |--api.html
  |--app.js
  |--config.js
  |--package.json
  |--package-lock.json

首先安装mongoose

$ npm install mongoose --save

在项目根目录下新建一个/models/mongo.js

var mongoose = require("mongoose");
var db = mongoose.connect('mongodb://node:p123456@127.0.0.1:27017/nodedb');
module.exports = db;

在app.js引入

var db = require('./models/mongo');

# 2.创建数据模型

在项目根目录下新建/models/user.js

var mongoose = require("mongoose");
//一个用户模型
var userSchema = new mongoose.Schema({
	username    : {type: String},
	password    : {type: String},
	isAdmin     : {type: Boolen},
	registerTime: {type: String}
});
//创建Model
var userModel = mongoose.model('User', userSchema );
module.exports = userModel;

这里先提一下，在这部分代码被引入和运行后，进入交互界面的nodedb数据库中，显示所有集合

> db
nodedb
> show collections
users

发现新增了一个名为users的集合，这是在创建数据模型时自动生成的

集合的命名遵从英文的复数规则，比如mongoose.model('User', userSchema )对应的集合名为users，而mongoose.model('Fish', userSchema )对应的集合名为fish

# 3.增删改查

这里简单提一下mongoose的增删改查，实际使用中会复杂很多，比如mongoose不建议直接使用update来更新数据

在routes中新建api.js

var user = require('../models/user');
var express = require('express');
var router = express.Router();
var sd = require('silly-datetime');//日期&时间模块，需npm install

router.get('/', function(req, res) {
	res.render('api');
});

在app.js中引入，并将/api的路由交由该模块处理

var apiRouter = require('./routes/api');
...
app.use('/api', apiRouter);

# 3.1 新增用户

回到api.js

router.post('/addUser', function (req, res) {
	//查找用户是否已存在
	user.findOne({username: req.query.username}, function (err, user) {
		if (err) {
			throw err;
		}
		if (user) {
			res.json({
				success: 0,
				message: 'User has already existed.'
			});
		}
	});
	//新建用户
	let newUser = new user();
	newUser.username = req.body.username;
	newUser.password = req.body.password;
	newUser.isAdmin = false;
	newUser.registerTime = sd.format(new Date(), 'YYYY-MM-DD HH:mm');
	newUser.save(function(err) {
		if (err) {
			res.send(err);
		}
		res.json({
			success: 1,
			message: 'User' + req.body.name + 'is created!'
		});
	});
});

删除用户

		});

# 3.4 查找用户

获取所有用户

router.get('/getUsers', function(req, res) {
	user.find(function(err, users) {
		if (err) {
			res.send(err);
		}
		res.json({
			success: 1,
			message: '',
			data: users
		});
	});
});

查找单个用户

router.get('/findUser', function(req, res) {
	user.findOne({name: req.query.name}, function (err, user) {
		if (err) {
			throw err;
		}
		if (!user) {
			res.json({
				success: 0,
				message: 'User is not found.'
			});
		}
		else if (user) {
			res.json({
				success: 1,
				message: '',
				data: user
			});
		}
	});
});
